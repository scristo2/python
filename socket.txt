import socket

####Solo se permite un uso de cada direcciÃ³n de socket -> si te sale este error,esque hay otro programa que esta usando ese puerto

mysocket = socket.socket(socket.AF_INET,socket.SOCK_STREAM)
mysocket.bind((socket.gethostname(),8000))
mysocket.listen(5)

while True:
    conexxion, addr = mysocket.accept()
    print('nueva conexion establecida')
    print(addr)
    peticion = conexxion.recv(102400)
    print(peticion.decode())
    message = "hola soy el servidor" 
    conexxion.send(message.encode())
mysocket.close()
